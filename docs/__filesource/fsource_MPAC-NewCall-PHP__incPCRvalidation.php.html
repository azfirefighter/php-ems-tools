  42. // this function handles all validation of the PCR form.
  43. // it should return an array that's empty or includes one or more of the following arrays:
  44. //  'errors'
  45. //  'warnings'
  46. // each array is another array, like InputFieldName => message
  47. function validate_PCR_form()
  48. {
  49.     global $PCRconfig_date_format_regex, $PCRconfig_time_format_regex;
  50.     $time_regex = $PCRconfig_time_format_regex;
  51.  
  52.     $ret = array('errors' => array(), 'warnings' => array());
  53.  
  54.     //$ret['errors']['Foo'] = "TODO - DEBUG - get rid of this error.";
  55.     //$ret['warnings']['Foo'] = "TODO - DEBUG - get rid of this warning.";
  56.  
  57.     //
  58.     // BEGIN VALIDATION
  59.     //
  60.  
  61.     // DATE
  62.     if(trim($_POST['Date']) == "" || ! preg_match($PCRconfig_date_format_regex, $_POST['Date']) || ! strtotime($_POST['Date']))
  63.     {
  64.     $ret['errors']['Date'] = "Date - Invalid date format.";
  65.     }
  66.     else
  67.     {
  68.     $_POST['ts_Date'] = strtotime($_POST['Date']);
  69.     }
  70.  
  71.     if($_POST['ts_Date'] > strtotime(date("Y-m-d ", time())." 00:00:00"))
  72.     {
  73.     $ret['errors']['Date'] = "Date cannot be in the future.";
  74.     }
  75.  
  76.  
  77.     // TIMES
  78.     // dispatch time - must be present and OK
  79.     $times_ts = array(); // for checking that times are in sequence
  80.     $min_time_so_far = ""; // the minimum time (not TS) found so far.
  81.     $changed_date = false; // did we have to change the date?
  82.     if(! isset($_POST['time_disp']) || trim($_POST['time_disp']) == "" || ! preg_match($time_regex, $_POST['time_disp']))
  83.     {
  84.     $ret['errors']['time_disp'] = "Dispatch Time - invalid time format or missing. <em>(if there was no actual dispatch, just use the in service time here as well.)</em>";
  85.     }
  86.     else
  87.     {
  88.     $_POST['ts_time_disp'] = strtotime($_POST['Date']." ".$_POST['time_disp']);
  89.     $times_ts[] = $_POST['ts_time_disp'];
  90.     $min_time_so_far = $_POST['time_disp'];
  91.     }
  92.  
  93.     // out of service time - must be present and OK
  94.     if(! isset($_POST['time_out']) || trim($_POST['time_out']) == "" || ! preg_match($time_regex, $_POST['time_out']))
  95.     {
  96.     $ret['errors']['time_out'] = "Out of Service Time - invalid time format or missing <em>(if you were dispatched to a second call, use the dispatch time for the second call as the out of service time for this call.)</em>.";
  97.     }
  98.  
  99.     // rest of times - check validity and make TS/add as appropriate
  100.     if(trim($_POST['time_insvc']) != "" || ! call_NoHosp()) { doTime("time_insvc", "In Service", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);}
  101.     if(trim($_POST['time_onscene']) != "") { doTime("time_onscene", "On Scene", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);}
  102.     if(call_transport()){ doTime("time_enroute", "EnRoute", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);}
  103.     if(call_transport()){ doTime("time_arrived", "Arrived", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);}
  104.     if(trim($_POST['time_onscene']) != "") { doTime("time_avail", "Available", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);}
  105.     doTime("time_out", "Out of Service", $time_regex, $ret, $times_ts, $min_time_so_far, $changed_date);
  106.  
  107.     // check that the times are in order
  108.     $foo = false;
  109.     for($i = 1; $i < count($times_ts); $i++)
  110.     {
  111.     if($times_ts[$i] < $times_ts[$i-1])
  112.     {
  113.         $foo = true;
  114.     }
  115.     }
  116.     if($foo)
  117.     {
  118.     $ret['errors']['times'] = "oops! Your times seem to be out of order.";
  119.     }
  120.  
  121.     // patient number
  122.     if(trim($_POST["patientNum"]) != "" && trim($_POST['patientOf']) != "" && (((int)$_POST["patientNum"]) > ((int)$_POST['patientOf'])))
  123.     {
  124.     $ret['errors']['patientNum'] = "You can not have a patient number higher than the total number of patients.";
  125.     }
  126.  
  127.     // call type
  128.     if(trim($_POST['CallType']) == "")
  129.     {
  130.     $ret['errors']['CallType'] = "You must select a call type.";
  131.     }
  132.  
  133.     // call location
  134.     if(trim($_POST["CallLoc"]) == "Other" && (! isset($_POST["calllocid"]) || trim($_POST["calllocid"]) == ""))
  135.     {
  136.     $ret['errors']['CallLoc'] = "If you selected 'Other' for a call location, you must select a location.";
  137.     }
  138.  
  139.     // make sure no driver to hospital if we didn't go to a hospital
  140.     if(isset($_POST["crew_driver_hosp"]) && call_NoHosp())
  141.     {
  142.     $ret['warnings']['crew_driver_hosp'] = "You cannot have a driver to the Hospital on a call without a patient transported. This has been automatically corrected.";
  143.     unset($_POST['crew_driver_hosp']);
  144.     }
  145.  
  146.     // check that there are no drivers on a call with no rig
  147.     if(isset($_POST["crew_driver_hosp"]) && call_NoRig())
  148.     {
  149.     $ret['warnings']['crew_driver_hosp'] = "You cannot have a driver to the hospital on a call where no ambulance was used. This has been automatically corrected.";
  150.     unset($_POST['crew_driver_hosp']);
  151.     }
  152.     if(isset($_POST["crew_driver_bldg"]) && call_NoRig())
  153.     {
  154.     $ret['warnings']['crew_driver_bldg'] = "You cannot have a driver to the building on a call where no ambulance was used. This has been automatically corrected.";
  155.     unset($_POST['crew_driver_bldg']);
  156.     }
  157.     if(isset($_POST["crew_driver_scene"]) && call_NoRig())
  158.     {
  159.     $ret['warnings']['crew_driver_scene'] = "You cannot have a driver to the scene on a call where no ambulance was used. This has been automatically corrected.";
  160.     unset($_POST['crew_driver_scene']);
  161.     }
  162.  
  163.     // equipment left
  164.     if(strtolower(trim($_POST["EquipLeft"])) == "none")
  165.     {
  166.     $ret['warnings']["EquipLeft"] = "If you left no equipment at the hospital, please leave the field blank. This has been automatically corrected.";
  167.     $_POST["EquipLeft"] = "";
  168.     }
  169.  
  170.     // passengers
  171.     if(strtolower(trim($_POST["Passengers"])) == "none")
  172.     {
  173.     $ret['warnings']["Passengers"] = "If you transported no passengers, please leave the field blank. This has been automatically corrected.";
  174.     $_POST["Passengers"] = "";
  175.     }
  176.  
  177.     // remarks
  178.     if(strlen(trim($_POST['remarks'])) < 8)
  179.     {
  180.     $ret['errors']['remarks'] = "You must enter a narrative for Remarks.";
  181.     }
  182.  
  183.     // other call type - OC
  184.     if(call_type_other() && trim($_POST['OC']) != "Other" && trim($_POST['OC']) != "NoCrew" && trim($_POST['OC']) != "Canceled")
  185.     {
  186.     $ret['errors']['OC'] = "You selected a call type that can only have an outcome of Other, No Crew, or Canceled.";
  187.     }
  188.  
  189.     // NoCrew but unit
  190.     if(trim($_POST['OC']) == "NoCrew" && trim($_POST['unit']) != "" && trim($_POST['unit']) != "N/A")
  191.     {
  192.     $ret['errors']['unit'] = "You cannot have a unit selected for a call with no crew.";
  193.     }
  194.  
  195.     // check mileage
  196.     if(trim($_POST['unit']) != "" && trim($_POST['unit']) != "N/A")
  197.     {
  198.     $foo = checkMileage(trim($_POST['unit']), trim($_POST['mileage']));
  199.     if($foo == "TOOHIGH"){ $ret['warnings']['mileage'] = "The mileage entered is 30+ miles more than the last recorded mileage for this rig. Are you sure it is correct? Perhaps you entered an incorrect mileage, or selected the wrong rig.";}
  200.     elseif($foo == "TOOLOW"){ $ret['errors']['mileage'] = "The mileage entered is less than the last recorded mileage for this rig. This is not possible. Please check the mileage and <strong>that you have selected the correct rig.</strong>";}
  201.     }
  202.  
  203.     // mileage but no unit
  204.     if(trim($_POST['mileage']) != "" && (trim($_POST['unit']) == "" || trim($_POST['unit']) == "N/A"))
  205.     {
  206.     $ret['errors']['mileage'] = "You cannot have a mileage for a call with no unit.";
  207.     }
  208.  
  209.     // unit but no mileage
  210.     if(trim($_POST['mileage']) == "" && trim($_POST['unit']) != "" && trim($_POST['unit']) != "N/A")
  211.     {
  212.     $ret['errors']['mileage'] = "You must enter a mileage for a call with a unit selected.";
  213.     }
  214.  
  215.     // unit but no driver
  216.     if((! isset($_POST['crew_driver_scene']) || ! isset($_POST['crew_driver_bldg'])) && trim($_POST['unit']) != "" && trim($_POST['unit']) != "N/A")
  217.     {
  218.     $ret['errors']['crew_driver_scene'] = "If you selected a unit, you must at least have a driver to the scene and building.";
  219.     }
  220.  
  221.     // signature
  222.     check_signature($ret);
  223.  
  224.     // check crew
  225.     checkCrew($ret);
  226.  
  227.     // OC or ALS implies transport, but no Transferred To
  228.     if(call_transport() && trim($_POST["PtTransTo"]) == "None")
  229.     {
  230.     $ret['errors']['PtTransTo'] = "You selected an Outcome or ALS status that implies transport, but failed to select a Transferred To location.";
  231.     }
  232.  
  233.     // ALS status implies a unit, but no unit specified
  234.     if(call_ALS() && trim($_POST['ALSunit']) == "")
  235.     {
  236.     $ret['errors']['ALSunit'] = "You selected an ALS status that should have an associated ALS unit, but you did not specify an ALS unit. <em>(hint: it's usually 402 in MP.)</em>";
  237.     }
  238.  
  239.     // transport but no interventions
  240.     if(call_transport() && trim($_POST["tx"]) == "")
  241.     {
  242.     $ret['errors']['tx'] = "You transported a patient. <em>Something</em> must be checked off for Treatments/Interventions. <em>(hint: Regular Stretcher)</em>.";
  243.     }
  244.  
  245.     // call type other but patient selected
  246.     if(call_type_other() && call_has_patient())
  247.     {
  248.     $ret['errors']['CallType'] = "A call type of Other cannot have a patient.";
  249.     }
  250.  
  251.     // call type other but patients home as location
  252.     if(call_type_other() && trim($_POST["CallLoc"]) == "Home")
  253.     {
  254.     $ret['errors']['CallLoc'] = "A call type of Other cannot have a call location of Patients Home.";
  255.     }
  256.  
  257.     // TODO - keep adding in validation rules here.
  258.  
  259.     // TODO - validate mileage against last known for that rig.
  260.  
  261.     //
  262.     // END VALIDATION
  263.     //
  264.  
  265.     if(count($ret['errors']) == 0){ unset($ret['errors']);}
  266.     if(count($ret['warnings']) == 0){ unset($ret['warnings']);}
  267.     return $ret;
  268. }

  270. function doTime($name, $title, $time_regex, &$ret, &$times_ts, &$min_time_so_far, &$changed_date)
  271. {
  272.  
  273.     // TODO - for outcomes that don't require certain times, go back and unset these errors
  274.     if(!isset($_POST[$name]) || trim($_POST[$name]) == "" || ! preg_match($time_regex, $_POST[$name]))
  275.     {
  276.     $ret['errors'][$name] = "$title Time - invalid time format.";
  277.     }
  278.     else
  279.     {
  280.     // the time is here, add it.
  281.     if(strtotime($_POST['Date']." ".$_POST[$name]) < strtotime($_POST['Date']." ".$min_time_so_far) || $changed_date)
  282.     {
  283.         // this time is before the last time. Assume we're changing the date forward a day
  284.         $min_time_so_far = $_POST[$name];
  285.         if(! $changed_date)
  286.         {
  287.         $ret['warnings'][$name] = "$title Time - this time is earlier in the day than the time before it, so the date has been changed to the NEXT day (assuming the call crossed over from one day to another). Please be SURE that the Date on the call is the date it was dispatched, NOT the date it was filled in.";
  288.         $changed_date = strtotime(date("Y-m-d", strtotime($_POST['Date'])+86400));
  289.         }
  290.         $_POST['ts_'.$name] = strtotime(date("Y-m-d", $changed_date)." ".$_POST[$name]);
  291.         $times_ts[] = $_POST['ts_'.$name];
  292.     }
  293.     else
  294.     {
  295.         $_POST['ts_'.$name] = strtotime($_POST['Date']." ".$_POST[$name]);
  296.         $times_ts[] = $_POST['ts_'.$name];
  297.     }
  298.     }
  299.  
  300. }

  307. function call_NoHosp()
  308. {
  309.     $foo = trim($_POST['OC']);
  310.     if($foo == "Canceled" || $foo == "Refusal" || $foo == "DOA" || $foo == "NoCrew")
  311.     {
  312.     return true;
  313.     }
  314.     return false;
  315. }

  317. function call_NoRig()
  318. {
  319.     if(trim($_POST['OC']) == "NoCrew" || trim($_POST['unit']) == "" || trim($_POST['unit']) == "N/A")
  320.     {
  321.     return true;
  322.     }
  323.     return false;
  324. }

  326. function call_type_other()

  328.     if(substr($_POST['CallType'], 0, 8) == "Other - ")
  329.     {
  330.     return true;

  335. function checkCrew(&$ret)
  336. {
  337.     $count = 0;
  338.     for($i = 0; $i < 12; $i++)
  339.     {
  340.     if(isset($_POST["crew_id_".$i]) && trim($_POST["crew_id_".$i]) != "")
  341.     {
  342.         $count++;
  343.     }
  344.     }
  345.     // edited 2010-04-19 by jantman, allow "NoCrew" calls without any crew members
  346.     if($count == 0 && $_POST['OC'] != "NoCrew")
  347.     {
  348.     $ret['errors']['crew_id_0'] = "You must specify at least one crew member unless call type is 'No Crew'.";
  349.     }
  350.  
  351.     require('config/crew.php');
  352.     for($i = 0; $i < 12; $i++)
  353.     {
  354.     if(isset($_POST["crew_id_".$i]) && trim($_POST["crew_id_".$i]) != "" && ! in_array(trim($_POST["crew_id_".$i]), $CREW_ARRAY))
  355.     {
  356.         $ret['errors']["crew_id_".$i] = "Crew $i (".trim($_POST["crew_id_".$i]).") is not a valid EMTid.";
  357.     }
  358.     }
  359. }

  361. function check_signature(&$ret)
  362. {
  363.     // no signature

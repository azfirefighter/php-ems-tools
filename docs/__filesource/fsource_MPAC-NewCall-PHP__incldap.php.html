  70. function authByLDAP($username, $password, $groupDN)
  71. {
  72.     /*
  73.      / @param username
  74.      / @param password
  75.      / @param groupDN - DN of group to require
  76.      / @param debug - whether to log debug info or not
  77.      /
  78.      / @return 0 on success, 1 on invalid credentials, 2 on group error, 3 on system error
  79.     */
  80.  
  81.     if($debug) { error_log("ldap_auth.php: Attempting LDAP Auth username=".$username);}

  83.     global $topdn, $debug;

  85.     $ds = ldap_connect();

  87.     if( !$ds )

  90.     return 3;
  91.     }
  92.  
  93.     if (! ldap_set_option($ds, LDAP_OPT_PROTOCOL_VERSION, 3))

  95.     if($debug) { error_log("ldap_auth.php: Failed to set protocol version to 3");}

  97.     }
  98.     
  99.     //Connection made -- bind anonymously and get dn for username.
  100.     $bind = @ldap_bind($ds);
  101.     
  102.     //Check to make sure we're bound.
  103.     if( !$bind )
  104.     {
  105.         if($debug) { error_log("ldap_auth.php: Anonymous bind to LDAP FAILED.  Contact Tech Services! (Debug 2)");}
  106.     return 3;
  107.     }
  108.     
  109.     $search = ldap_search($ds, $topdn, "uid=$username"); // do the search
  110.     if($debug){ error_log("ldap_auth.php: searcing established ds with topdn: $topdn search: 'uid=$username'");}
  111.  
  112.     //Make sure only ONE result was returned -- if not, they might've thrown a * into the username.  Bad user!
  113.     if( ldap_count_entries($ds,$search) != 1 )

  123.  
  124.     //Now, try to rebind with their full dn and password.
  125.     $bind = @ldap_bind($ds, $info[0]['dn'], $password);
  126.     if( !$bind || !isset($bind))
  127.     {
  128.     if($debug) { error_log("ldap_auth.php: AUTH failed: ".ldap_error($ds)." (".ldap_errno($ds).")");}
  129.     return 1;
  130.     }
  131.     
  132.     //Now verify the previous search using their credentials.

  135.     $info = ldap_get_entries($ds, $search);

  190. function getGroupMembers($ds, $groupDN)
  191. {
  192.     if(is_string($groupDN))
  193.     {
  194.     $ri = ldap_read($ds, $groupDN, "objectClass=*");
  195.     if(! $ri){ return false;}
  196.     $foo = ldap_get_entries($ds, $ri);
  197.     $foo = $foo[0]['member'];
  198.     if(isset($foo['count'])){ unset($foo['count']);}
  199.     }
  200.  
  201.     $recurse = array();
  202.     foreach($foo as $key => $val)
  203.     {
  204.     if(strstr($val, "ou=groups")){ $recurse[] = $val;}
  205.     }
  206.     
  207.     // else is array
  208.     $result = array();
  209.     foreach($recurse as $key => $val)
  210.     {
  211.     $ri = ldap_read($ds, $val, "objectClass=*");
  212.     if(! $ri){ return false;}
  213.     $foo = ldap_get_entries($ds, $ri);
  214.     $foo = $foo[0]['member'];
  215.     if(isset($foo['count'])){ unset($foo['count']);}
  216.     $result = array_merge($result, $foo);
  217.     }
  218.     return $result;
  219. }

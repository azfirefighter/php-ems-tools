  42. /**
  43.  * Tells if a row from the table is different from an array of new values.
  44.  * @param string $pkey_name the name of the table's primary key column (will be ignored)
  45.  * @param Array $oldRow the row returned by a MySQL query (the current data)
  46.  * @param Array $newArr the new row (like above but with new data)
  47.  * @return boolean true if the arrays are different, false if they are the same
  48.  * @author Jason Antman <jason@jasonantman.com>
  49.  */
  50. function PCR_arrays_differ($pkey_name, $oldRow, $newArr, $ignoreKeys = null)
  51. {
  52.     $DIFF_DEBUG = true; // TODO - disable once we know things are working OK
  53.  
  54.     if($DIFF_DEBUG){ echo '<pre>'; echo "===orig newArr=== count=".count($newArr)."\n"; echo var_dump($newArr); echo '</pre>';}
  55.     $newArr = array_merge($oldRow, $newArr);
  56.     if($DIFF_DEBUG){ echo '<pre>'; echo "===new newArr=== count=".count($newArr)."\n"; echo var_dump($newArr); echo '</pre>';}
  57.  
  58.     // strip out the stuff that doesn't matter/should be different
  59.     if(isset($oldRow['Pkey'])){ unset($oldRow['Pkey']);}
  60.     if(isset($newArr['Pkey'])){ unset($newArr['Pkey']);}
  61.     if(isset($oldRow['is_from_old_calls'])){ unset($oldRow['is_from_old_calls']);}
  62.     if(isset($newArr['is_from_old_calls'])){ unset($newArr['is_from_old_calls']);}
  63.     if(isset($oldRow['is_deprecated'])){ unset($oldRow['is_deprecated']);}
  64.     if(isset($newArr['is_deprecated'])){ unset($newArr['is_deprecated']);}
  65.     if(isset($oldRow['deprecated_by_id'])){ unset($oldRow['deprecated_by_id']);}
  66.     if(isset($newArr['deprecated_by_id'])){ unset($newArr['deprecated_by_id']);}
  67.     if(isset($oldRow['deprecates_id'])){ unset($oldRow['deprecates_id']);}
  68.     if(isset($newArr['deprecates_id'])){ unset($newArr['deprecates_id']);}
  69.     if(isset($oldRow['submit_ts'])){ unset($oldRow['submit_ts']);}
  70.     if(isset($newArr['submit_ts'])){ unset($newArr['submit_ts']);}
  71.     if(isset($oldRow['update_ts'])){ unset($oldRow['update_ts']);}
  72.     if(isset($newArr['update_ts'])){ unset($newArr['update_ts']);}
  73.     if(isset($oldRow[$pkey_name])){ unset($oldRow[$pkey_name]);}
  74.     if(isset($newArr[$pkey_name])){ unset($newArr[$pkey_name]);}
  75.  
  76.     foreach($oldRow as $key => $val)
  77.     {
  78.     if($oldRow[$key] == null && ! isset($newArr[$key])){ unset($oldRow[$key]);}
  79.     }
  80.  
  81.     foreach($newArr as $key => $val)
  82.     {
  83.     if($newArr[$key] == null && ! isset($oldRow[$key])){ unset($newArr[$key]);}
  84.     }
  85.  
  86.     if($ignoreKeys != null)
  87.     {
  88.     foreach($ignoreKeys as $key => $val)
  89.     {
  90.         if(isset($oldRow[$key])){ unset($oldRow[$key]);}
  91.         if(isset($newArr[$key])){ unset($newArr[$key]);}
  92.     }
  93.     }
  94.  
  95.     if($DIFF_DEBUG){ echo '<pre>===array_diff_assoc output:\n'; echo var_dump(array_diff_assoc($oldRow, $newArr)); echo "\n======OLD=====\n"; echo var_dump($oldRow); echo "\n=====NEW====\n"; echo var_dump($newArr); echo '</pre>'; echo '<p>count(oldRow)='.count($oldRow).' count(newRow)='.count($newArr).'</p>';}
  96.  
  97.     if(count(array_diff_assoc($oldRow, $newArr)) > 0){ return true;}
  98.  
  99.     return false;
  100. }

  102. /**
  103.  * determines whether or not to update a given row
  104.  *
  105.  * This does a query for the specified row of the specified table using
  106.  * WHERE $pkey_name=$pkey_val AND is_deprecated=0
  107.  * and then calls PCR_arrays_differ() to compare the resulting row with
  108.  * the newArr array. If they differ, it returns true, if not it returns false.
  109.  *
  110.  * @param string $table the name of the table the row is found in
  111.  * @param string $pkey_name the name of the primary key field
  112.  * @param int $pkey_val the value of the primary key field for this function
  113.  * @param Array $newArr the array matching the MySQL result row with the (possibly) new values
  114.  * @return boolean true if the rows differ and it needs to be updated, false otherwise
  115.  */
  116. function PCR_row_needs_update($table, $pkey_name, $pkey_val, $newArr, $ignoreKeys = null)
  117. {
  118.     $query = "SELECT * FROM $table WHERE $pkey_name=$pkey_val AND is_deprecated=0;";
  119.     $result = mysql_query($query) or db_error($query, mysql_error());
  120.     $row = mysql_fetch_assoc($result);
  121.     return PCR_arrays_differ($pkey_name, $row, $newArr);
  122. }

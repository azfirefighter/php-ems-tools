  54. function is_valid_session()
  55. {
  56.     $query = "SELECT * FROM web_sessions WHERE sessid='".session_id()."';";
  57.     $result = mysql_query($query);
  58.     if(mysql_num_rows($result) < 1){ return 1;}
  59.     $row = mysql_fetch_assoc($result);
  60.     if(time() >= $row['expire_ts']) { kill_session(); return 3;}
  61.     if(! isset($row['EMTid'])){ return 2;}
  62.     if(! isset($_SESSION['EMTid'])){ return 2;}
  63.     if($_SERVER['REMOTE_ADDR'] != $row['remote_ip']){ kill_session(); return 2;}
  64.     if($row['last_load_ts'] < time() - SESS_LOAD_TIMEOUT){ kill_session(); return 1;}
  65.     return 0;
  66. }

  68. function login_start_sess($userid, $pass)
  69. {
  70.     if(ldap_auth_handler($userid, $pass) == false)
  71.     {    
  72.     error_log("login_start_sess: ldap_auth_handler returned False for ($userid)");
  73.     return false;
  74.     }
  75.     $query = "INSERT INTO web_sessions SET sessid='".session_id()."',start_ts=".time().",last_load_ts=".time().",EMTid='".mysql_real_escape_string($userid)."',remote_ip='".mysql_real_escape_string($_SERVER['REMOTE_ADDR'])."',user_agent='".mysql_real_escape_string($_SERVER['HTTP_USER_AGENT'])."',expire_ts=".(time() + SESS_TIMEOUT).";";
  76.     $result = mysql_query($query);
